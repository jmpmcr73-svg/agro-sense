    <script>

        const imageUpload = document.getElementById('image-upload');

        const imagePreview = document.getElementById('image-preview');

        const previewImage = document.getElementById('preview-image');

        const diagnoseBtn = document.getElementById('diagnose-btn');

        const loadingSpinner = document.getElementById('loading-spinner');

        const resultContainer = document.getElementById('result-container');

        const diagnosisOutput = document.getElementById('diagnosis-output');

 

        // Escucha el cambio de archivo para mostrar la vista previa

        imageUpload.addEventListener('change', (event) => {

            const file = event.target.files[0];

            if (file) {

                const reader = new FileReader();

                reader.onload = (e) => {

                    previewImage.src = e.target.result;

                    imagePreview.classList.remove('hidden');

                };

                reader.readAsDataURL(file);

            } else {

                imagePreview.classList.add('hidden');

            }

        });

 

        // Escucha el clic del botón de diagnóstico

        diagnoseBtn.addEventListener('click', async () => {

            const imageFile = imageUpload.files[0];

            const ph = document.getElementById('ph').value;

            const ce = document.getElementById('ce').value;

            const tds = document.getElementById('tds').value;

            const salinity = document.getElementById('salinity').value;

            const chlorophyll = document.getElementById('chlorophyll').value;

            const nitrogen = document.getElementById('nitrogen').value;

            const notes = document.getElementById('notes').value;

 

            if (!imageFile && !notes) {

                diagnosisOutput.innerHTML = '<p class="text-red-500">Por favor, sube una foto o proporciona notas adicionales.</p>';

                resultContainer.classList.remove('hidden');

                return;

            }

 

            loadingSpinner.classList.remove('hidden');

            resultContainer.classList.add('hidden');

           

            const base64Image = await new Promise((resolve) => {

                if (imageFile) {

                    const reader = new FileReader();

                    reader.onload = () => resolve(reader.result.split(',')[1]);

                    reader.readAsDataURL(imageFile);

                } else {

                    resolve(null);

                }

            });

 

            // Construir el prompt para la IA

            const textPrompt = `Eres un agrónomo experto. Realiza un diagnóstico de la salud de una planta y proporciona recomendaciones.

           

            **Datos Proporcionados:**

            - pH: ${ph || 'No proporcionado'}

            - CE: ${ce || 'No proporcionada'}

            - TDS: ${tds || 'No proporcionado'}

            - Salinidad: ${salinity || 'No proporcionada'}

            - Clorofila en hoja: ${chlorophyll || 'No proporcionada'}

            - Nitrógeno en hoja: ${nitrogen || 'No proporcionado'}

            - Notas adicionales: ${notes || 'No proporcionadas'}

           

            **Analiza la foto y la información. Luego, genera una respuesta estructurada que incluya:**

            1. Un **Diagnóstico General** (posibles plagas, enfermedades, o deficiencias).

            2. **Deficiencias de Elementos Macros y Micros** (si aplica).

            3. **Recomendaciones Específicas** (cómo corregir los problemas).`;

 

            try {

                const payload = {

                    contents: [{

                        role: "user",

                        parts: [

                            { text: textPrompt },

                            ...(base64Image ? [{ inlineData: { mimeType: imageFile.type, data: base64Image } }] : [])

                        ]

                    }],

                };

 

                const apiKey = "";

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

 

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

 

                const result = await response.json();

                const generatedText = result?.candidates?.[0]?.content?.parts?.[0]?.text;

 

                if (generatedText) {

                    // Formatear el texto de salida con etiquetas HTML

                    const formattedText = generatedText

                        .replace(/\*\*(.*?)\*\*/g, '<h3>$1</h3>')

                        .replace(/^- (.*?)<br\s*>/g, '<ul><li>$1</li></ul>')

                        .replace(/\n/g, '<br>');

                    diagnosisOutput.innerHTML = formattedText;

                } else {

                    diagnosisOutput.innerHTML = '<p class="text-red-500">No se pudo obtener un diagnóstico. Por favor, intenta de nuevo.</p>';

                }

 

            } catch (error) {

                console.error("Error al llamar a la API:", error);

                diagnosisOutput.innerHTML = '<p class="text-red-500">Ocurrió un error. Por favor, verifica tu conexión o intenta más tarde.</p>';

            } finally {

                loadingSpinner.classList.add('hidden');

                resultContainer.classList.remove('hidden');

            }

        });

    </script>
